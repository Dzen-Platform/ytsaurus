SET(YT_VERSION_MINOR 3)


SET(PROTO_NAMESPACE yt)

SET(YT_ARCADIA_CXXFLAGS
    -Wno-sign-compare
    -Wno-unused-function
    -Wno-unused-parameter
    -Wno-unused-value
    -Wno-uninitialized
    -Wno-type-limits
    -Wno-missing-braces
    -Wno-missing-field-initializers
    -Wunused-result
)

IF(GCC)
    SET(YT_ARCADIA_CXXFLAGS
        ${YT_ARCADIA_CXXFLAGS}
        -Wno-maybe-uninitialized
    )
ENDIF()

SET(YT_ABI_VERSION "${YT_VERSION_MAJOR}_${YT_VERSION_MINOR}")

DEFAULT(YT_VERSION_PATCH 0)
