SET(YT_VERSION_MAJOR 20)
SET(YT_VERSION_MINOR 2)

INCLUDE(${ARCADIA_ROOT}/yt/ya.make.environment.inc)

SET(PROTO_NAMESPACE ${YT_ROOT})

SET(YT_ARCADIA_CXXFLAGS
    -Wno-sign-compare
    -Wno-unused-function
    -Wno-unused-parameter
    -Wno-unused-value
    -Wno-uninitialized
    -Wno-type-limits
    -Wno-missing-braces
    -Wno-missing-field-initializers
    -Wunused-result
    -DYT_IN_ARCADIA
)

SET(YT_ARCADIA_CFLAGS
    -D__SSE4_2__
    -msse4.2
    -mpclmul
)

IF(AUTOCHECK)
    SET(YT_ARCADIA_CXXFLAGS
        ${YT_ARCADIA_CXXFLAGS}
        -DAUTOCHECK
    )
ENDIF()

IF(GCC)
    SET(YT_ARCADIA_CXXFLAGS
        ${YT_ARCADIA_CXXFLAGS}
        -Wno-maybe-uninitialized
    )
ENDIF()

SET(YT_ABI_VERSION "${YT_VERSION_MAJOR}_${YT_VERSION_MINOR}")

DEFAULT(YT_VERSION_PATCH 0)
DEFAULT(YT_VERSION_BRANCH "local")
